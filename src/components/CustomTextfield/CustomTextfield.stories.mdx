# CustomTextfield

This is a demo component, will be removed before release to production.

## Renderer

Below is an interactive example, please check the form data output in the console log in browser's inspector.

<Canvas>
  <Story name="Renderer">
    {() =>
      createForm({
        form: {
          components: [
            {
              type: "customtextfield",
              key: "customtextfield",
              label: "Custom textfield label",
            },
            {
              type: "button",
              key: "submit",
              label: "Submit",
              action: "submit",
            },
          ],
        },
        controller: (formio) => {
          formio.on("submit", (e) => {
            console.info("submit", e.data);
          });
          formio.on("change", (e) => {
            console.info("change", e.data);
          });
        },
      })
    }
  </Story>
</Canvas>
